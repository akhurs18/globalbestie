Google Sheets ConnectionLegacy / Optional
Create a blank Google Sheet. Rename the first tab to Orders and add this header row: Date | Name | Phone | Email | City | Address | Items | Total | Advance | Status | Notes
The script auto-creates the Products tab and Orders tab headers if missing.
Go to Extensions → Apps Script, paste the script below, deploy as Web App (Execute as: Me, Who has access: Anyone), and copy the URL here.
⚠️ Share your Google Sheet: click Share → Anyone with the link → Viewer so orders can be read without login.

function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();

  // Products tab
  if (e.parameter.type === 'products') {
    var ps = ss.getSheetByName('Products');
    if (!ps || ps.getLastRow() < 2)
      return ContentService.createTextOutput(JSON.stringify([]))
        .setMimeType(ContentService.MimeType.JSON);
    var pd = ps.getDataRange().getValues();
    var ph = pd[0];
    var rows = pd.slice(1).map(function(row) {
      var obj = {}; ph.forEach(function(h,i){ obj[h]=row[i]; }); return obj;
    });
    return ContentService.createTextOutput(JSON.stringify(rows))
      .setMimeType(ContentService.MimeType.JSON);
  }

  // Orders tab
  var sheet = ss.getSheetByName('Orders');
  if (!sheet || sheet.getLastRow() < 2)
    return ContentService.createTextOutput(JSON.stringify([]))
      .setMimeType(ContentService.MimeType.JSON);
  var data = sheet.getDataRange().getValues();
  var headers = data[0];
  var rows = data.slice(1).map(function(row) {
    var obj = {};
    headers.forEach(function(h, i) { obj[h] = row[i]; });
    return obj;
  });
  return ContentService
    .createTextOutput(JSON.stringify(rows))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var raw = e.parameter.payload || e.postData.contents;
  var data = JSON.parse(raw);

  if (data.action === 'setProducts') {
    var ps = ss.getSheetByName('Products');
    if (!ps) ps = ss.insertSheet('Products');
    ps.clearContents();
    ps.appendRow(['id','gender','cat','name','brand','pkr','inStock','description','image']);
    data.products.forEach(function(p) {
      ps.appendRow([p.id,p.gender,p.cat,p.name,p.brand,p.pkr,
                    p.inStock?'true':'false', p.description||'', p.image||'']);
    });
  } else {
    var sheet = ss.getSheetByName('Orders');
    if (!sheet) {
      sheet = ss.insertSheet('Orders');
      sheet.appendRow(['Date','Name','Phone','Email','City','Address','Items','Total','Advance','Status','Notes']);
    }
    if (data.action === 'addOrder') {
      sheet.appendRow([
        data.date, data.name, data.phone, data.email,
        data.city, data.address, data.items,
        data.total, data.advance, 'New Order', data.notes
      ]);
    } else if (data.action === 'updateStatus') {
      var rows = sheet.getDataRange().getValues();
      for (var i = 1; i < rows.length; i++) {
        if (rows[i][2] == data.phone && rows[i][0] == data.date) {
          sheet.getRange(i+1, 10).setValue(data.status);
          break;
        }
      }
    }
  }
  return ContentService
    .createTextOutput(JSON.stringify({success: true}))
    .setMimeType(ContentService.MimeType.JSON);
}
